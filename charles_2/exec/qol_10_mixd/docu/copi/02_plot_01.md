# 1. Forest plot of *modeled means only*

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# Extract modeled means
params = result.params
vcov = result.cov_params()

# Compute modeled means and CIs
def modeled_mean_and_ci(level):
    if level == "T0":
        est = params["Intercept"]
        var = vcov.loc["Intercept", "Intercept"]
    else:
        coef = f"C(timepoint)[T.{level}]"
        est = params["Intercept"] + params[coef]
        var = (
            vcov.loc["Intercept", "Intercept"]
            + vcov.loc[coef, coef]
            + 2 * vcov.loc["Intercept", coef]
        )
    se = np.sqrt(var)
    ci_low = est - 1.96 * se
    ci_up = est + 1.96 * se
    return est, ci_low, ci_up

timepoints = ["T0", "T1", "T2"]
estimates = [modeled_mean_and_ci(tp) for tp in timepoints]

df_modl = pd.DataFrame(estimates, columns=["mean", "ci_lower", "ci_upper"])
df_modl["timepoint"] = timepoints

# Forest plot
plt.figure(figsize=(6, 4))
y_pos = np.arange(len(df_modl))

plt.errorbar(
    df_modl["mean"],
    y_pos,
    xerr=[df_modl["mean"] - df_modl["ci_lower"],
          df_modl["ci_upper"] - df_modl["mean"]],
    fmt="o",
    color="black",
    ecolor="gray",
    capsize=4,
    markersize=6,
)

plt.yticks(y_pos, df_modl["timepoint"])
plt.xlabel("VEINES-QOL T-score")
plt.title("Modeled VEINES-QOL Means with 95% CI")
plt.grid(axis="x", linestyle="--", alpha=0.4)

plt.tight_layout()
plt.show()
```
